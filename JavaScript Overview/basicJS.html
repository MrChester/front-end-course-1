Object

Object is an unordered dynamic collection of properties.

key - unique string

value - primitive value, object, or function

Object


    var x = {
        a: 10, // primitive value
        b: { z: 100 }, // object "b" with property z
        c: function () { // function (method)
            alert('method x.c');
        }
    };

    alert(x.a); // 10
    alert(x.b.z); // 100
    x.c(); // 'method x.c'

Dynamic nature


    var foo = { x: 10 };

    foo.y = 20; // add new property
    console.log(foo); // {x: 10, y: 20}

    // change property value to function
    foo.x = function () {
        console.log('foo.x');
    };

    foo.x(); // 'foo.x'

    delete foo.x;  // delete property
    console.log(foo); // {y: 20}

For..In statement

Iterate through all of the members of an object:


    for (name in object) {
        if (object.hasOwnProperty(name)) {
        // within the loop,
        // name is the key of current member
        // object[name] is the current value
            console.log("name: " + name);
            console.log("value: " + object[name]);
        }
    }

For..In statement


    // monkey patching

    var foo = {moo: 2};

    for(var i in foo) {

Object methods (ES3)


    var foo = { x: 10, y: 20 };
    console.log( foo.constructor == Object );
    console.log( foo.toString() == "[object Object]" );
    console.log( foo.valueOf() == foo );
    console.log( foo.hasOwnProperty("x") == true );
    console.log( foo.hasOwnProperty("toString") == false );
    console.log( foo.propertyIsEnumerable("x") == true );
    console.log( Object.prototype.isPrototypeOf(foo) == true );

Native and host objects

Native - defined by the ECMAScript specification. (Array, Date or user-defined (var o = {};))

Host - defined by the host environment (window and all the DOM objects)

    // monkey patching
    Function.prototype.method = function(name, fn) {
        this.prototype[name] = fn;
    };

    function Car(model, color) {
        this.model = model;
        this.color = color;
    };

    Car.method("accelerate", function (effort) { /* ... */ });
    Car.method("brake", function (effort) { /* ... */ });

    var lada = new Car("Kalina", "Gray");
